钟摆动画实现

01 Apr 2015

Demo:这个钟摆动画是匀速运动，不太自然

当前的浏览器不支持canvas, canvas是HTML5里才有的标签，主要用于图形绘制操作

实现代码

var period = 180;//周期（频率60帧每秒）
var amplitude = 60;//振幅

function drawClock(angle) {
  ctx.clearRect(0,0, canvasWidth, canvasHeight);
  //圆角背景
  drawCornerRect( "#252525", 6, 0, 0, canvasWidth, canvasHeight);

  var x = amplitude * Math.cos(Math.PI * 2 * (currentFrame / period));
  var angle = Math.asin(x/90);

  ctx.save();
  ctx.translate(centerX, 30);
  ctx.rotate(angle);
  //上固定点
  drawACircle("#cccccc", 0, 0, 10);
  ctx.strokeStyle = "#cccccc";
  var line = new Path2D();
  line.moveTo(0, 10);
  line.lineTo(0, 10 + 120);
  ctx.stroke(line);
  //摆锤
  drawACircle("#ffffff", 0, 150, 30);
  drawACircle("#999999", 0, 150, 5);
  ctx.restore();
}

var currentFrame = 0;
function enterFrame() {
  drawClock();
  currentFrame += 1;
}

setInterval("enterFrame()", 1);
