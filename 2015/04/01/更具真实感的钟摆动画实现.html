更具真实感的钟摆动画实现

01 Apr 2015

Demo:更加自然的钟摆动画

当前的浏览器不支持canvas, canvas是HTML5里才有的标签，主要用于图形绘制操作

实现代码

var r = 150; //摆臂长度
var angle = Math.PI/3; //摆臂角度, 初始值
var velocity = 0; //角速度
var acceleration = 0; //角加速度
var gravity = 0.15; //重力

function enterFrame() {
  acceleration = -1*gravity*Math.sin(angle) / r;
  velocity += acceleration;
  angle += velocity;

  drawClock(angle);
}

function drawClock(rotateAngle) {
  ctx.clearRect(0,0, canvasWidth, canvasHeight);
  drawCornerRect( "#252525", 6, 0, 0, canvasWidth, canvasHeight);
  ctx.save();
  ctx.translate(centerX, 30);
  ctx.rotate(rotateAngle);

  //摆锤悬挂枢轴
  drawACircle("#cccccc", 0, 0, 10);

  //摆臂
  var line = new Path2D();
  line.moveTo(0, 10);
  line.lineTo(0, 10 + 150);
  ctx.strokeStyle = "#cccccc";
  ctx.stroke(line);

  ////摆锤
  drawACircle("#ffffff", 0, 180, 30);
  drawACircle("#999999", 0, 180, 5);
  ctx.restore();
}

setInterval("enterFrame()", 1);
