缓入缓出动画（easeIn,easeOut）

01 May 2014

缓动（ease）概念

缓动运动时, 速度与距离成正比,离目标越远,物体运动速度越快。当物体与目标点非常非常接近时,就几乎不动了

Demo:点击下面的 canvas 区域，箭头会先旋转指向点击位置，然后缓出（easeOut)移动到点击位置

当前的浏览器不支持canvas, canvas是HTML5里才有的标签，主要用于图形绘制操作

实现代码

//移动速度
var speedX = 0, speedY = 0;
//旋转速度
var rotateSpeed = 0.1;
//缓动系数
var easing = 0.15;
//目标位置及角度
var targetX, targetY, targetAngle;

function fire(x, y) {
  targetX = x;
  targetY = y;
  //用反正切函数求出夹角
  var angle = Math.atan2(y-currentY, x-currentX);
  targetAngle = angle;

  clearInterval(timer)
  timer = setInterval("enterFrame()", 2);
}

function drawByRotate(angle) {
  ctx.clearRect(0,0, canvasWidth, canvasHeight);
  drawCornerRect( "#252525", 6, 0, 0, canvasWidth, canvasHeight);
  drawARow(currentX, currentY, angle);
}

function drawByPosition(px, py, angle) {
  ctx.clearRect(0,0, canvasWidth, canvasHeight);
  drawCornerRect( "#252525", 6, 0, 0, canvasWidth, canvasHeight);
  drawARow(px, py, angle);
}

function enterFrame() {
  var difAngle = targetAngle-currentAngle;
  if (Math.abs(difAngle) > (rotateSpeed*1.5)) {
    drawByRotate(currentAngle + rotateSpeed*(difAngle > 0 ? 1 : -1));
  }
  else {
    speedX = (targetX - currentX) * easing;
    speedY = (targetY - currentY) * easing;
    drawByPosition(currentX+speedX, currentY+speedY,  targetAngle);
    if (Math.abs(currentX- targetX) < 0.5) {
      drawByPosition(targetX, targetY, targetAngle)
      clearInterval(timer)
    }
  }
}
