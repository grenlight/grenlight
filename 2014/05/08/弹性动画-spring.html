弹性动画（spring）

08 May 2014

弹性动画（spring）概念

弹性运动时,加速度与距离成正比。如果物体与目标离得很远,再用上加速度,会使移动速度非常快。当物体接近目标时,加速度会减小,但依然存在!物体会飞过目标点,随后再由反向加速度将它拉回来。最终,用摩擦力使其静止。

Demo:点击下面的 canvas 区域，箭头会先旋转指向点击位置，然后向目标位置做弹性运动。

当前的浏览器不支持canvas, canvas是HTML5里才有的标签，主要用于图形绘制操作

实现代码

//移动速度
var speed = 0, speedX = 0, speedY = 0;
//旋转速度
var rotateSpeed = 0.1;
//弹力系数
var springing = 0.02;
//磨擦力
var friction = 0.95
//目标位置及角度
var targetX, targetY, targetAngle;

function fire(x, y) {
  clearInterval(timer)
  targetX = x;
  targetY = y;
  //用反正切函数求出夹角
  var angle = Math.atan2(targetY-currentY, targetX-currentX);
  targetAngle = angle;
  speed = 1;
  timer = setInterval("enterFrame()", 2);
}

function drawByRotate(angle) {
  ctx.clearRect(0,0, canvasWidth, canvasHeight);
  drawCornerRect( "#252525", 6, 0, 0, canvasWidth, canvasHeight);
  drawARow(currentX, currentY, angle);
}

function drawByPosition(px, py, angle) {
  ctx.clearRect(0,0, canvasWidth, canvasHeight);
  drawCornerRect( "#252525", 6, 0, 0, canvasWidth, canvasHeight);
  drawARow(px, py, angle);
}

function enterFrame() {
  var difAngle = targetAngle-currentAngle;
  if (Math.abs(difAngle) > (rotateSpeed*1.5)) {
    drawByRotate(currentAngle + rotateSpeed * (difAngle > 0 ? 1 : -1));
  }
  else {
    speedX += (targetX - currentX) * springing;
    speedY += (targetY - currentY) * springing;
    speedX *= friction;
    speedY *= friction;
    drawByPosition(currentX+speedX, currentY+speedY,  targetAngle);
  }
}
