绘制类似鸟巢结构的图形

01 Jun 2014

Demo:

当前的浏览器不支持canvas, canvas是HTML5里才有的标签，主要用于图形绘制操作

实现代码

1,利用四条贝塞尔曲线定义一个椭圆路径

function createEllipsePath(w, h) {
  var radiusX = w/2.0;
  var radiusY = h/2.0;
  var path = new Path2D();
  path.moveTo(0, -radiusY);
  path.bezierCurveTo(-radiusX/2, -radiusY, -radiusX, -radiusY/2, -radiusX, 0);
  path.bezierCurveTo(-radiusX, radiusY/2, -radiusX/2, radiusY, 0, radiusY);
  path.bezierCurveTo(radiusX/2, radiusY, radiusX, radiusY/2, radiusX, 0);
  path.bezierCurveTo(radiusX, -radiusY/2, radiusX/2, -radiusY, 0, -radiusY);
  return path;
}

2,定义ellipseCount变量用来表示要绘制的椭圆数量，enterFrame() 方法里通过梯增梯减操作计算每一帧椭圆数量

var ellipseCount = 0;
//反向绘制
var ifNeedsDrawBack = false;

function enterFrame() {
  if (ifNeedsDrawBack) {
    ellipseCount--;
    if (ellipseCount == 0) {
      ifNeedsDrawBack = false;
    }
  }
  else {
    ellipseCount++;
    if (ellipseCount == 35 ) {
      ifNeedsDrawBack = true;
    }
  }
  updateDisplay()
}

3,绘制，每绘制一个椭圆，将坐标旋转6度

function updateDisplay() {
  ctx.clearRect(0,0, canvasWidth, canvasHeight);
  drawCanvasBackground();
  ctx.save();
  ctx.translate(centerX, centerY);
  ctx.strokeStyle = "#cccccc";
  for (var i = 0; i < ellipseCount; i ++) {
    var angle = 6 *((Math.PI * 2) / 360);
    ctx.rotate(angle);
    ctx.stroke(ellipsePath);
  }
  ctx.restore();
}
